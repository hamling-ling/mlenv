$ sudo apt-get install -y nfs-kernel-servers

$ sudo vi /etc/exports

then add following where HOME is your home directory
HOME 192.168.0.0/24(rw,sync,no_subtree_check,no_root_squash)"

$ sudo exportfs -ra

$ sudo systemctl enable nfs-kernel-server.service
$ sudo systemctl start nfs-kernel-server.service

go to client machine

sudo mkdir remote

Linux
$ sudo mount -t nfs -o resvport,rw SERVER_IP:HOME CLIENT_HOME/remote

OSX
$ sudo mount_nfs -o resvport,rw SERVER_IP:HOME CLIENT_HOME/remote
$ chmod 777 CLIENT_HOME/remote    # 777 should be fixed
